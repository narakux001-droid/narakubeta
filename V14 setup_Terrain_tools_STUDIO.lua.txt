local function makeDraggable(frame, handle)
	local h = handle or frame
	local dragging, dragStart, startPos = false, nil, nil
	h.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true; dragStart = input.Position; startPos = frame.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then dragging = false end
			end)
		end
	end)
	h.InputChanged:Connect(function(input)
		if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and dragging then
			local d = input.Position - dragStart
			frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset+d.X, startPos.Y.Scale, startPos.Y.Offset+d.Y)
		end
	end)
end

-- ═══════════════════════════════════════════════════════════
-- VOLUME & SIZE ACCURATE
-- ═══════════════════════════════════════════════════════════
local function calcVolume(shape, sx, sy, sz)
	if shape=="Bola" then local r=sx/2; return (4/3)*math.pi*r*r*r
	elseif shape=="Silinder" then local r=math.max(sx,sz)*0.6/2; return math.pi*r*r*sy
	elseif shape=="Baji" or shape=="Sudut" then return sx*sy*sz*0.5
	else return sx*sy*sz end
end
local function getAccurateSize(shape, sx, sy, sz)
	if shape=="Bola" then local r=sx/2; return Vector3.new(r*2,r*2,r*2),r
	elseif shape=="Silinder" then return Vector3.new(sx,sy,sz),math.max(sx,sz)*0.6
	else return Vector3.new(sx,sy,sz),nil end
end

local function applyMaterialColor(T, matName, matE)

local function doTerrainFill(T, pos, size, shape, matE, rot)
	local cf=CFrame.new(pos)*CFrame.Angles(0,math.rad(rot or 0),0)
	local accSize,accRadius=getAccurateSize(shape,size.X,size.Y,size.Z)
	if shape=="Bola" then T:FillBall(pos,accRadius,matE)
	elseif shape=="Silinder" then T:FillCylinder(cf*CFrame.Angles(0,0,math.rad(90)),size.Y,accRadius,matE)
	else T:FillBlock(cf,accSize,matE) end
end

local function applyTerrainForPart(part, matName)
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end
	if not terrainOnlyMaterials[matName] then return end
	local matE=materialEnum[matName]; if not matE then return end
	pcall(function()
		local sz=part.Size
		local shapeStr=part:IsA("WedgePart") and "Baji" or
			(part.Shape==Enum.PartType.Ball and "Bola" or
			(part.Shape==Enum.PartType.Cylinder and "Silinder" or "Balok"))
		local accSize,accRadius=getAccurateSize(shapeStr,sz.X,sz.Y,sz.Z)
		if part.Shape==Enum.PartType.Ball then T:FillBall(part.Position,accRadius,matE)
		elseif part.Shape==Enum.PartType.Cylinder then T:FillCylinder(part.CFrame*CFrame.Angles(0,0,math.rad(90)),sz.Y,accRadius,matE)
		else T:FillBlock(part.CFrame,accSize,matE) end
		applyMaterialColor(T,matName,matE)
	end)
end

local function applyTerrainDirectly(pos, size, shape, matName, rotDeg)
	local T=Workspace:FindFirstChild("Terrain"); if not T then return nil end
	if not terrainOnlyMaterials[matName] then return nil end
	local matE=materialEnum[matName]; if not matE then return nil end
	if not _TB.terrainDataByMaterial[matName] then _TB.terrainDataByMaterial[matName]={} end
	local isColorable=colorableMaterials[matName] and true or false
	local sR,sG,sB=0,0,0
	if isColorable then sR,sG,sB=getActiveColorRGB(matName) end
	local rot=rotDeg or 0
	local vol=calcVolume(shape,size.X,size.Y,size.Z)
	local entry={position=pos,size=size,shape=shape,material=matName,isColorable=isColorable,grR=sR,grG=sG,grB=sB,rotation=rot,volume=vol,_undone=false}
	table.insert(_TB.terrainDataByMaterial[matName],entry)
	pcall(function() doTerrainFill(T,pos,size,shape,matE,rot); applyMaterialColor(T,matName,matE) end)
	return entry
end

local function convertPartsToTerrain()
	local T=Workspace:FindFirstChild("Terrain"); if not T then return 0 end
	local c=0
	for _,matName in ipairs(materials) do
		if terrainOnlyMaterials[matName] then
			local f=Workspace:FindFirstChild(matName)
			if f then for _,part in ipairs(f:GetChildren()) do
				if part:IsA("BasePart") then pcall(function() applyTerrainForPart(part,matName);c=c+1 end) end
			end end
		end
	end
	return c
end

-- ═══════════════════════════════════════════════════════════
-- [A1] SEA LEVEL — isi Water di bawah Y target
-- Data disimpan ke _TB.seaLevelData agar masuk konversi script
-- ═══════════════════════════════════════════════════════════
local function applySeaLevel(targetY, matName, areaHalfSize)
	local T=Workspace:FindFirstChild("Terrain"); if not T then return false end
	local matE=materialEnum[matName] or Enum.Material.Water
	local stepSize=8
	local R=areaHalfSize or 200
	local isColorable=colorableMaterials[matName] and true or false
	local grR,grG,grB=0,0,0
	if isColorable then grR,grG,grB=getActiveColorRGB(matName) end
	pcall(function()
		for dx=-R,R,stepSize do
			for dz=-R,R,stepSize do
				-- Isi dari bawah (targetY-areaHalfSize) sampai targetY
				local fillPos=Vector3.new(dx,targetY-stepSize/2,dz)
				local fillSize=Vector3.new(stepSize,stepSize,stepSize)
				T:FillBlock(CFrame.new(fillPos),fillSize,matE)
			end
			task.wait()
		end
		applyMaterialColor(T,matName,matE)
	end)
	-- Simpan ke _TB.seaLevelData untuk konversi
	table.insert(_TB.seaLevelData,{
		y=targetY, matName=matName, areaSize=R,
		isColorable=isColorable, grR=grR, grG=grG, grB=grB,
		stepSize=stepSize
	})
	return true
end

-- ═══════════════════════════════════════════════════════════
-- [A2] REPLACE MATERIAL
-- Ganti semua terrain material X → Y
-- Data disimpan ke _TB.replaceData agar masuk konversi script
-- ═══════════════════════════════════════════════════════════
local function doReplaceMaterial(fromMat, toMat)
	local T=Workspace:FindFirstChild("Terrain"); if not T then return 0 end
	local fromEnum=materialEnum[fromMat]; local toEnum=materialEnum[toMat]
	if not fromEnum or not toEnum then return 0 end
	local count=0
	-- Replace di _TB.terrainDataByMaterial (data brush)
	if _TB.terrainDataByMaterial[fromMat] then
		-- Pindahkan semua entry dari fromMat ke toMat
		if not _TB.terrainDataByMaterial[toMat] then _TB.terrainDataByMaterial[toMat]={} end
		for _,entry in ipairs(_TB.terrainDataByMaterial[fromMat]) do
			if not entry._undone then
				entry.material=toMat
				local isColorable=colorableMaterials[toMat] and true or false
				entry.isColorable=isColorable
				if isColorable then entry.grR,entry.grG,entry.grB=getActiveColorRGB(toMat)
				else entry.grR,entry.grG,entry.grB=0,0,0 end
				table.insert(_TB.terrainDataByMaterial[toMat],entry)
				-- Re-fill di terrain
				pcall(function()
					doTerrainFill(T,entry.position,entry.size,entry.shape,toEnum,entry.rotation)
					applyMaterialColor(T,toMat,toEnum)
				end)
				count=count+1
			end
		end
		_TB.terrainDataByMaterial[fromMat]={}
	end
	-- Simpan operasi replace ke _TB.replaceData untuk konversi
	local isColorable=colorableMaterials[toMat] and true or false
	local grR,grG,grB=0,0,0
	if isColorable then grR,grG,grB=getActiveColorRGB(toMat) end
	table.insert(_TB.replaceData,{fromMat=fromMat,toMat=toMat,isColorable=isColorable,grR=grR,grG=grG,grB=grB})
	return count
end

local function updateTitleCounter()
	if _TB.titleLabel then
		local n=getTotalTerrainCount()

			and ("Terrain Builder V14  •  ["..n.."/"..TERRAIN_LIMIT.."]")

	end

local function pushUndo(entry)
	table.insert(_TB.brushUndoStack,entry)
	if #_TB.brushUndoStack>150 then table.remove(_TB.brushUndoStack,1) end
	_TB.brushRedoStack={}
end

local function undoBrushTerrain()
	if #_TB.brushUndoStack==0 then return end
	local last=table.remove(_TB.brushUndoStack)
	table.insert(_TB.brushRedoStack,last)
	local T=Workspace:FindFirstChild("Terrain")
	if last.type=="swipe" then
		local data=_TB.swipeData[last.index]
		if data then
			data._undone=true
			if T then pcall(function()
				local cf=CFrame.new(data.position)*CFrame.Angles(0,math.rad(data.rotation or 0),0)
				T:FillBlock(cf,data.size,Enum.Material.Air)
			end) end
		end
	elseif last.type=="terrain" then
		local matName=last.matName; local idx=last.index
		local data=_TB.terrainDataByMaterial[matName] and _TB.terrainDataByMaterial[matName][idx]
		if data then
			data._undone=true
			if T then pcall(function()
				local cf=CFrame.new(data.position)*CFrame.Angles(0,math.rad(data.rotation or 0),0)
				if data.shape=="Bola" then T:FillBall(data.position,data.size.X/2,Enum.Material.Air)
				elseif data.shape=="Silinder" then T:FillCylinder(cf*CFrame.Angles(0,0,math.rad(90)),data.size.Y,math.max(data.size.X,data.size.Z)*0.6,Enum.Material.Air)
				else T:FillBlock(cf,data.size,Enum.Material.Air) end
			end) end
			if _TB.limitReached and getTotalTerrainCount()<TERRAIN_LIMIT then
				_TB.limitReached=false
				if UI.brushTerrainOnlyBtn and _TB.brushTerrainOnly then
					UI.brushTerrainOnlyBtn.Text="Tanpa Part: ON ["..getTotalTerrainCount().."/"..TERRAIN_LIMIT.."]"
				end
			end
		end
	end
	updateTitleCounter()
end

local function redoBrushTerrain()
	if #_TB.brushRedoStack==0 then return end
	local entry=table.remove(_TB.brushRedoStack)
	table.insert(_TB.brushUndoStack,entry)
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end
	if entry.type=="swipe" then
		local data=_TB.swipeData[entry.index]
		if data then data._undone=false; pcall(function() doTerrainFill(T,data.position,data.size,data.shape,materialEnum[data.material],data.rotation); applyMaterialColor(T,data.material,materialEnum[data.material]) end) end
	elseif entry.type=="terrain" then
		local data=_TB.terrainDataByMaterial[entry.matName] and _TB.terrainDataByMaterial[entry.matName][entry.index]
		if data then data._undone=false; pcall(function() doTerrainFill(T,data.position,data.size,data.shape,materialEnum[data.material],data.rotation); applyMaterialColor(T,data.material,materialEnum[data.material]) end) end
	end
	updateTitleCounter()
end

local function brushTerrainOnly_Pure(pos)
	if _TB.limitReached then return end
	if not terrainOnlyMaterials[_TB.currentMaterial] then return end
	local t=tick(); if t-_TB.lastBrushTime<_TB.brushCooldown then return end
	local minDist=_TB.currentShape=="Bola" and (_TB.currentBlockSize.X/2)*1.2 or math.min(_TB.width,_TB.height,_TB.depth)*0.9
	if _TB.lastBrushPos and (pos-_TB.lastBrushPos).Magnitude<minDist then return end
	if getTotalTerrainCount()>=TERRAIN_LIMIT then
		_TB.limitReached=true;_TB.brushTerrainOnly=false
		if UI.brushTerrainOnlyBtn then UI.brushTerrainOnlyBtn.Text="Tanpa Part: OFF";UI.brushTerrainOnlyBtn.BackgroundColor3=Color3.fromRGB(75,58,95) end
		return
	end
	_TB.lastBrushTime=t;_TB.lastBrushPos=pos
	local entry=applyTerrainDirectly(pos,_TB.currentBlockSize,_TB.currentShape,_TB.currentMaterial,_TB.rotationAngle)
	if entry then
		local idx=#_TB.terrainDataByMaterial[_TB.currentMaterial]
		pushUndo({type="terrain",matName=_TB.currentMaterial,index=idx,data=entry})
		if UI.brushTerrainOnlyBtn then UI.brushTerrainOnlyBtn.Text="Tanpa Part: ON ["..getTotalTerrainCount().."/"..TERRAIN_LIMIT.."]" end
		updateTitleCounter()
	end
end

local function brushTerrainContinuous(pos)
	if not terrainOnlyMaterials[_TB.currentMaterial] then return end
	if _TB.lastBrushPos and (pos-_TB.lastBrushPos).Magnitude<(math.min(_TB.width,_TB.height,_TB.depth)*0.3) then return end
	_TB.lastBrushPos=pos
	local folder=getFolder(_TB.currentMaterial)
	local part
	if _TB.currentShape=="Bola" then part=Instance.new("Part");part.Shape=Enum.PartType.Ball
	elseif _TB.currentShape=="Silinder" then part=Instance.new("Part");part.Shape=Enum.PartType.Cylinder
		part.CFrame=CFrame.new(pos)*CFrame.Angles(0,math.rad(_TB.rotationAngle),0)*CFrame.Angles(0,0,math.rad(90))
	elseif _TB.currentShape=="Baji" or _TB.currentShape=="Sudut" then part=Instance.new("WedgePart")
	else part=Instance.new("Part") end
	part.Size=_TB.currentBlockSize;part.Material=materialEnum[_TB.currentMaterial] or Enum.Material.Plastic
	part.Color=materialColors[_TB.currentMaterial] or Color3.new(1,1,1)
	part.Anchored=true;part.CanCollide=true;part.CastShadow=false
	if _TB.currentShape~="Silinder" then part.CFrame=CFrame.new(pos)*CFrame.Angles(0,math.rad(_TB.rotationAngle),0) end
	if _TB.currentMaterial=="Water" then part.Transparency=0.5;part.CanCollide=false
	elseif _TB.currentMaterial=="Ice" or _TB.currentMaterial=="Glacier" then part.Transparency=0.3 end
	part.Parent=folder
	table.insert(_TB.brushTrailParts,part)
	if _TB.autoConvertToTerrain then applyTerrainForPart(part,_TB.currentMaterial) end
	pcall(function() part:SetNetworkOwner(nil) end)
end

local function createTunnel(pos)
	if not pos then return end
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end
	local cf=CFrame.new(pos)*CFrame.Angles(0,math.rad(_TB.rotationAngle),0)
	if _TB.currentShape=="Bola" then T:FillBall(pos,_TB.currentBlockSize.X/2,Enum.Material.Air)
	elseif _TB.currentShape=="Silinder" then T:FillCylinder(cf*CFrame.Angles(0,0,math.rad(90)),_TB.currentBlockSize.Y,math.max(_TB.currentBlockSize.X,_TB.currentBlockSize.Z)*0.6,Enum.Material.Air)
	else T:FillBlock(cf,_TB.currentBlockSize,Enum.Material.Air) end
	local r=math.max(_TB.width,_TB.height,_TB.depth)*0.7
	for _,matName in ipairs(materials) do
		local f=Workspace:FindFirstChild(matName)
		if f then for _,p in ipairs(f:GetChildren()) do if p:IsA("BasePart") and (p.Position-pos).Magnitude<r then p:Destroy() end end end
	end
end

local function brushSwipe(pos)
	if not pos then return end
	local t=tick(); if t-_TB.lastSwipeBrushTime<_TB.swipeBrushCooldown then return end
	if _TB.lastSwipePos and (pos-_TB.lastSwipePos).Magnitude<(math.min(_TB.width,_TB.height,_TB.depth)*0.4) then return end
	_TB.lastSwipeBrushTime=t;_TB.lastSwipePos=pos
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end
	local targetMat=materialEnum[_TB.currentMaterial]; if not targetMat then return end
	local swipeSize=Vector3.new(_TB.width,math.max(_TB.height,8),_TB.depth)
	local fillPos=Vector3.new(pos.X,pos.Y-(swipeSize.Y*0.5),pos.Z)
	local cf=CFrame.new(fillPos)*CFrame.Angles(0,math.rad(_TB.rotationAngle),0)
	local vol=calcVolume(_TB.currentShape,swipeSize.X,swipeSize.Y,swipeSize.Z)
	pcall(function()
		if _TB.currentShape=="Bola" then T:FillBall(pos,swipeSize.X/2,targetMat)
		elseif _TB.currentShape=="Silinder" then T:FillCylinder(cf*CFrame.Angles(0,0,math.rad(90)),swipeSize.Y,math.max(swipeSize.X,swipeSize.Z)*0.6,targetMat)
		else T:FillBlock(cf,swipeSize,targetMat) end
		applyMaterialColor(T,_TB.currentMaterial,targetMat)
	end)
	local isColorable=colorableMaterials[_TB.currentMaterial] and true or false
	local r,g,b=0,0,0
	if isColorable then r,g,b=getActiveColorRGB(_TB.currentMaterial) end
	table.insert(_TB.swipeData,{position=fillPos,size=swipeSize,shape=_TB.currentShape,material=_TB.currentMaterial,isColorable=isColorable,grR=r,grG=g,grB=b,rotation=_TB.rotationAngle,volume=vol,_undone=false})
	pushUndo({type="swipe",index=#_TB.swipeData})
end

local function brushTree(pos)
	if not pos then return end
	local t=tick(); if t-_TB.lastTreeBrushTime<_TB.treeBrushCooldown then return end
	_TB.lastTreeBrushTime=t
	local treeFolder=Workspace:FindFirstChild("POHON"); if not treeFolder then warn("⚠ Folder 'POHON' tidak ditemukan!"); return end
	local originals={}
	for _,v in ipairs(treeFolder:GetChildren()) do
		if not v:GetAttribute("PohonClone") then table.insert(originals,v) end
	end
	if #originals==0 then return end
	local clone=originals[math.random(1,#originals)]:Clone()
	clone:SetAttribute("PohonClone",true)
	local randRot=math.random(0,360)
	if clone:IsA("Model") then
		local ok,cf,sz=pcall(function() return clone:GetBoundingBox() end)
		if ok and cf and sz then
			clone:PivotTo(CFrame.new(pos.X,pos.Y+(cf.Position.Y-sz.Y/2),pos.Z)*CFrame.Angles(0,math.rad(randRot),0))
		else clone:PivotTo(CFrame.new(pos)) end
	elseif clone:IsA("BasePart") then
		clone.CFrame=CFrame.new(pos)*CFrame.Angles(0,math.rad(randRot),0);clone.Anchored=true
	end
	clone.Parent=treeFolder
end

-- Special brushes
local function brushFlatten(pos)
	if not pos then return end
	local t=tick(); if t-_TB.lastSpecialBrushTime<_TB.specialBrushCooldown then return end; _TB.lastSpecialBrushTime=t
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end
	local matE=materialEnum[_TB.currentMaterial] or Enum.Material.Grass
	local r=math.max(_TB.width,_TB.depth)*(_TB.brushStrength*0.5); local step=4
	pcall(function()
		for dx=-r,r,step do for dz=-r,r,step do
			if dx*dx+dz*dz<=r*r then
				T:FillBlock(CFrame.new(pos.X+dx,_TB.flattenTargetY+step*2,pos.Z+dz),Vector3.new(step,step*4,step),Enum.Material.Air)
				T:FillBlock(CFrame.new(pos.X+dx,_TB.flattenTargetY,pos.Z+dz),Vector3.new(step,step,step),matE)
			end
		end end
		applyMaterialColor(T,_TB.currentMaterial,matE)
	end)
end
local function brushSmooth(pos)
	if not pos then return end
	local t=tick(); if t-_TB.lastSpecialBrushTime<_TB.specialBrushCooldown then return end; _TB.lastSpecialBrushTime=t
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end
	local r=math.max(_TB.width,_TB.depth)*(_TB.brushStrength*0.4); local step=4
	local matE=materialEnum[_TB.currentMaterial] or Enum.Material.Grass
	pcall(function()
		for _,off in ipairs({Vector3.new(step,0,0),Vector3.new(-step,0,0),Vector3.new(0,0,step),Vector3.new(0,0,-step)}) do
			T:FillBall((pos+pos+off*(r/step))*0.5,step*0.8,matE)
		end
		applyMaterialColor(T,_TB.currentMaterial,matE)
	end)
end
local function brushErode(pos)

	pcall(function() T:FillBall(pos,math.max(_TB.width,_TB.depth)*(_TB.brushStrength*0.5),Enum.Material.Air) end)
end
local function brushGrow(pos)
	if not pos then return end
	local t=tick(); if t-_TB.lastSpecialBrushTime<_TB.specialBrushCooldown then return end; _TB.lastSpecialBrushTime=t

end
local function brushPaint(pos)
	if not pos then return end
	local t=tick(); if t-_TB.lastSpecialBrushTime<_TB.specialBrushCooldown then return end; _TB.lastSpecialBrushTime=t
	local T=Workspace:FindFirstChild("Terrain"); if not T then return end

	local matE=materialEnum[_TB.currentMaterial]; if not matE then return end

	pcall(function()
		if _TB.currentShape=="Bola" then T:FillBall(pos,r,matE)
		else T:FillBlock(CFrame.new(pos)*CFrame.Angles(0,math.rad(_TB.rotationAngle),0),Vector3.new(r*2,_TB.height,r*2),matE) end
		applyMaterialColor(T,_TB.currentMaterial,matE)
	end)
end

-- GENERATE SCRIPT — SYNC SEMUA OPERASI (brush+sealevel+replace)
-- ═══════════════════════════════════════════════════════════
local function makeScriptHeader(totalData, totalVol, part, totalParts)
	local partInfo=totalParts>1 and (" | Bagian "..part.."/"..totalParts) or ""
	local L={}
	table.insert(L,"-- ════════════════════════════════════════")
	table.insert(L,"-- Terrain Script | BY SYGKMUNII V14"..partInfo)
	table.insert(L,"-- Total titik  : "..totalData)
	table.insert(L,string.format("-- Total volume : %.1f studs³",totalVol))
	if totalParts>1 then table.insert(L,"-- ⚠ Jalankan semua bagian berurutan!") end
	table.insert(L,"-- ════════════════════════════════════════")
	table.insert(L,"local T=workspace.Terrain")
	table.insert(L,"local M={[1]=Enum.Material.Grass,[2]=Enum.Material.Ground,[3]=Enum.Material.Mud,")
	table.insert(L,"[4]=Enum.Material.Sand,[5]=Enum.Material.Snow,[6]=Enum.Material.LeafyGrass,")
	table.insert(L,"[7]=Enum.Material.Rock,[8]=Enum.Material.Slate,[9]=Enum.Material.Basalt,")
	table.insert(L,"[10]=Enum.Material.Cobblestone,[11]=Enum.Material.Granite,[12]=Enum.Material.Limestone,")
	table.insert(L,"[13]=Enum.Material.Pavement,[14]=Enum.Material.Salt,[15]=Enum.Material.CrackedLava,")
	table.insert(L,"[16]=Enum.Material.Marble,[17]=Enum.Material.Water,[18]=Enum.Material.Ice,[19]=Enum.Material.Glacier}")
	table.insert(L,"-- sh:1=Balok 2=Bola 3=Baji 4=Silinder 5=Sudut")
	table.insert(L,"local D={")
	return table.concat(L,"\n")
end
local function makeScriptFooter(seaBlocks, replaceOps)
	local L={}
	table.insert(L,"}")
	table.insert(L,"for i,d in ipairs(D)do")
	table.insert(L," local m=M[d.m]")
	table.insert(L," local cf=CFrame.new(d.p)*CFrame.Angles(0,math.rad(d.rot or 0),0)")
	table.insert(L," if d.sh==2 then T:FillBall(d.p,d.sx/2,m)")
	table.insert(L," elseif d.sh==4 then T:FillCylinder(cf*CFrame.Angles(0,0,math.rad(90)),d.sy,math.max(d.sx,d.sz)*0.6,m)")
	table.insert(L," else T:FillBlock(cf,Vector3.new(d.sx,d.sy,d.sz),m) end")
	table.insert(L," if d.r then T:SetMaterialColor(m,Color3.fromRGB(d.r,d.g,d.b)) end")
	table.insert(L," if i%20==0 then task.wait() end")
	table.insert(L,"end")
	-- [A1] Sea Level blocks dalam footer
	if seaBlocks and #seaBlocks>0 then
		table.insert(L,"-- === SEA LEVEL ===")
		for _,sl in ipairs(seaBlocks) do
			local mnStr=sl.matName
			table.insert(L,string.format(
				"do local slM=Enum.Material.%s; local slR=%d; local slStep=%d",
				mnStr, sl.areaSize, sl.stepSize))
			table.insert(L,string.format(
				" local slY=%.2f", sl.y))
			table.insert(L," for dx=-slR,slR,slStep do for dz=-slR,slR,slStep do")
			table.insert(L,"  T:FillBlock(CFrame.new(dx,slY-slStep/2,dz),Vector3.new(slStep,slStep,slStep),slM)")
			table.insert(L," end task.wait() end")
			if sl.isColorable then
				table.insert(L,string.format(
					" T:SetMaterialColor(slM,Color3.fromRGB(%d,%d,%d))",
					sl.grR,sl.grG,sl.grB))
			end
			table.insert(L,"end")
		end
	end
	-- [A2] Replace operations dalam footer
	if replaceOps and #replaceOps>0 then
		table.insert(L,"-- === REPLACE MATERIAL ===")
		for _,rp in ipairs(replaceOps) do
			table.insert(L,string.format(
				"-- Note: Replace %s→%s sudah diterapkan pada data D di atas",
				rp.fromMat, rp.toMat))
		end
	end
	return table.concat(L,"\n")
end
local function entryToLine(e, isLast)
	local comma=isLast and "" or ","
	local mn=materialToNumber[e.material] or 1
	local sn=(e.shape=="Balok"and 1)or(e.shape=="Bola"and 2)or(e.shape=="Baji"and 3)or(e.shape=="Silinder"and 4)or 5
	local rot=e.rotation or 0
	if e.isColorable then
		return string.format("{p=Vector3.new(%.2f,%.2f,%.2f),sx=%.2f,sy=%.2f,sz=%.2f,m=%d,sh=%d,r=%d,g=%d,b=%d,rot=%d}%s",
			e.position.X,e.position.Y,e.position.Z,e.size.X,e.size.Y,e.size.Z,mn,sn,e.grR,e.grG,e.grB,rot,comma)
	else
		return string.format("{p=Vector3.new(%.2f,%.2f,%.2f),sx=%.2f,sy=%.2f,sz=%.2f,m=%d,sh=%d,rot=%d}%s",
			e.position.X,e.position.Y,e.position.Z,e.size.X,e.size.Y,e.size.Z,mn,sn,rot,comma)
	end
end

local function generateScriptChunks()
	local allData,totalData={},0
	for _,list in pairs(_TB.terrainDataByMaterial) do
		for _,d in ipairs(list) do if not d._undone then table.insert(allData,d);totalData=totalData+1 end end
	end
	for _,d in ipairs(_TB.swipeData) do
		if not d._undone then table.insert(allData,d);totalData=totalData+1 end
	end
	if totalData==0 and #_TB.seaLevelData==0 then return nil end
	local totalVol=0
	for _,e in ipairs(allData) do totalVol=totalVol+(e.volume or 0) end
	local dataLines={}
	for i,e in ipairs(allData) do table.insert(dataLines,entryToLine(e,i==#allData)) end
	local HEADER_EST,FOOTER_EST=800,600
	local DATA_BUDGET=SCRIPT_CHAR_LIMIT-HEADER_EST-FOOTER_EST
	-- Bagi data ke chunks
	local chunks={}
	local curLines,curSize={},0
	for _,line in ipairs(dataLines) do
		local ls=#line+1
		if curSize+ls>DATA_BUDGET and #curLines>0 then
			table.insert(chunks,curLines);curLines={};curSize=0
		end
		table.insert(curLines,line);curSize=curSize+ls
	end
	if #curLines>0 then table.insert(chunks,curLines) end
	if #chunks==0 then table.insert(chunks,{}) end  -- minimal 1 chunk untuk sea level
	local totalChunks=#chunks
	local result={}
	for ci,lines in ipairs(chunks) do
		local isLast=ci==totalChunks
		local header=makeScriptHeader(totalData,totalVol,ci,totalChunks)
		-- Sea level dan replace hanya di chunk terakhir
		local footer=makeScriptFooter(isLast and _TB.seaLevelData or {}, isLast and _TB.replaceData or {})
		local dataStr=table.concat(lines,"\n")
		local full=header.."\n"..dataStr.."\n"..footer
		table.insert(result,{content=full,part=ci,totalParts=totalChunks,charCount=#full})
	end
	return result
end

-- ═══════════════════════════════════════════════════════════
local function getNextTerrainIndex()
	local i=1
	while i<=9999 do
		local exists=pcall(function() readfile(HASIL_FOLDER.."/Terrain"..i..".txt") end)
		if not exists then return i end
		i=i+1
	end
	return i
end

local function saveChunksToStorage(chunks)
	safeMakeFolder(DELTA_WORKSPACE);safeMakeFolder(HASIL_FOLDER)
	local baseIndex=getNextTerrainIndex()
	local savedFiles={}
	if #chunks==1 then
		local fn="Terrain"..baseIndex..".txt"
		local fp=HASIL_FOLDER.."/"..fn
		local ok,err=safeWriteFile(fp,chunks[1].content)
		table.insert(savedFiles,{ok=ok,path=fp,name=fn,err=err})
	else
		for _,chunk in ipairs(chunks) do
			local fn="Terrain"..baseIndex.."_part"..chunk.part..".txt"
			local fp=HASIL_FOLDER.."/"..fn
			local ok,err=safeWriteFile(fp,chunk.content)
			table.insert(savedFiles,{ok=ok,path=fp,name=fn,err=err})
		end
	end
	return savedFiles
end

	safeMakeFolder(DELTA_WORKSPACE);safeMakeFolder(HASIL_FOLDER)
	safeWriteFile(AUTOSAVE_PATH,chunks[1].content)
end

local function updatePreview(pos)
	local anyActive=_TB.placementEnabled or _TB.brushTerrainOnly or _TB.swipeModeActive or _TB.tunnelMode
		or _TB.flattenMode or _TB.smoothMode or _TB.erodeMode or _TB.growMode or _TB.paintMode or _TB.seaLevelMode
	if not anyActive or not pos then
		previewBox.Adornee=nil
		previewPart.Transparency=1
		return
	end

	-- Tentukan warna berdasarkan mode
	local col
	if _TB.erodeMode then col=Color3.fromRGB(255,60,60)
	elseif _TB.tunnelMode then col=Color3.fromRGB(255,140,0)
	elseif _TB.flattenMode then col=Color3.fromRGB(80,200,255)
	elseif _TB.smoothMode then col=Color3.fromRGB(220,220,0)
	elseif _TB.growMode then col=Color3.fromRGB(0,255,80)
	elseif _TB.paintMode then col=materialColors[_TB.currentMaterial] or Color3.fromRGB(200,100,255)
	elseif _TB.seaLevelMode then col=Color3.fromRGB(0,150,255)
	else col=materialColors[_TB.currentMaterial] or Color3.fromRGB(0,220,255) end

	previewBox.Color3=col
	previewBox.SurfaceColor3=col

	-- Update ukuran & posisi part (invisible, hanya sebagai anchor SelectionBox)
	previewPart.Size=_TB.currentBlockSize
	previewPart.Transparency=1
	local cf=CFrame.new(pos)*CFrame.Angles(0,math.rad(_TB.rotationAngle),0)
	if _TB.currentShape=="Bola" then
		previewPart.Shape=Enum.PartType.Ball
		previewPart.CFrame=cf
	elseif _TB.currentShape=="Silinder" then
		previewPart.Shape=Enum.PartType.Cylinder
		previewPart.CFrame=cf*CFrame.Angles(0,0,math.rad(90))
	else
		previewPart.Shape=Enum.PartType.Block
		previewPart.CFrame=cf
	end
	previewBox.Adornee=previewPart
end

local function refreshFileList()
	for _,c in ipairs(UI.fileListScroll:GetChildren()) do
		if not c:IsA("UIListLayout") then c:Destroy() end
	end

	local count=0

		local name=fp:match("([^/\\]+)$") or fp
		if name:match("%.txt$") and not name:match("_autosave") then
			count=count+1

				if content then pcall(function() setclipboard(content) end)
					row.Text="✅ Di-copy! — "..name
					task.wait(2);row.Text="📄 "..name
				end
			end)
		end
	end
	if count==0 then

		empty.Size=UDim2.new(1,-6,0,22);empty.BackgroundTransparency=1

-- ═══════════════════════════════════════════════════════════
local function refreshStats()
	local totalN=getTotalTerrainCount()
	local totalVol=0
	local perMat={}
	for _,list in pairs(_TB.terrainDataByMaterial) do
		for _,d in ipairs(list) do
			if not d._undone then
				totalVol=totalVol+(d.volume or 0)
				perMat[d.material]=(perMat[d.material] or 0)+1
			end
		end
	end
	local lines={}
	table.insert(lines,"Total titik : "..totalN.."/"..TERRAIN_LIMIT)
	table.insert(lines,string.format("Total volume: %.0f studs³",totalVol))
	table.insert(lines,"Swipe data  : "..#_TB.swipeData)
	table.insert(lines,"Sea level   : "..#_TB.seaLevelData.." ops")
	table.insert(lines,"Replace     : "..#_TB.replaceData.." ops")
	table.insert(lines,"Undo stack  : "..#_TB.brushUndoStack)
	local top={}
	for m,c in pairs(perMat) do table.insert(top,{m=m,c=c}) end
	table.sort(top,function(a,b) return a.c>b.c end)
	for i=1,math.min(3,#top) do
		table.insert(lines,(materialNames[top[i].m] or top[i].m)..": "..top[i].c)
	end
	UI.statsLabel.Text=table.concat(lines,"\n")
end
_TB.refreshStats = refreshStats

local function _setupInput()
-- ═══════════════════════════════════════════════════════════
local function getSnappedPos(hit)
	if not hit then return nil end
	local pos=hit.Position
	if _TB.lockY then pos=Vector3.new(pos.X,_TB.lockedY,pos.Z) end
	local g=_TB.currentShape=="Bola" and math.max(2,math.floor(math.min(_TB.width,_TB.height,_TB.depth)/2)) or math.min(_TB.width,_TB.height,_TB.depth)
	g=math.max(g,1)
	return Vector3.new(math.floor(pos.X/g+0.5)*g,math.floor(pos.Y/g+0.5)*g,math.floor(pos.Z/g+0.5)*g)
end

mouse.Button1Down:Connect(function()
	_TB.isMouseDown=true
	if not mouse or not mouse.Hit then return end
	local pos=getSnappedPos(mouse.Hit); if not pos then return end
	if _TB.treeBrushReady and _TB.treeBrushActive then brushTree(pos)
	elseif _TB.swipeMode and _TB.swipeModeActive then _TB.lastSwipePos=nil;brushSwipe(pos)
	elseif _TB.tunnelMode then createTunnel(pos)
	elseif _TB.flattenMode then brushFlatten(pos)
	elseif _TB.smoothMode then brushSmooth(pos)
	elseif _TB.erodeMode then brushErode(pos)
	elseif _TB.growMode then brushGrow(pos)
	elseif _TB.paintMode then brushPaint(pos)
	elseif _TB.brushTerrainOnly and _TB.placementEnabled then _TB.brushTrailParts={};_TB.lastBrushPos=nil;brushTerrainOnly_Pure(pos)
	elseif _TB.placementEnabled then _TB.brushTrailParts={};_TB.lastBrushPos=nil;brushTerrainContinuous(pos)
	end
end)
mouse.Button1Up:Connect(function() _TB.isMouseDown=false;_TB.lastBrushPos=nil;_TB.lastSwipePos=nil end)

RunService.Heartbeat:Connect(function()
	if not _TB.isMouseDown or not mouse or not mouse.Hit then return end
	local pos=getSnappedPos(mouse.Hit); if not pos then return end
	if _TB.treeBrushReady and _TB.treeBrushActive then brushTree(pos)
	elseif _TB.swipeMode and _TB.swipeModeActive then brushSwipe(pos)
	elseif _TB.flattenMode then brushFlatten(pos)
	elseif _TB.smoothMode then brushSmooth(pos)
	elseif _TB.erodeMode then brushErode(pos)
	elseif _TB.growMode then brushGrow(pos)
	elseif _TB.paintMode then brushPaint(pos)
	elseif _TB.placementEnabled then
		if _TB.brushTerrainOnly then brushTerrainOnly_Pure(pos) else brushTerrainContinuous(pos) end
	end
	-- [B1] Auto-save check
	local now=tick()
	if now-_TB.lastAutoSave>AUTOSAVE_INTERVAL then _TB.lastAutoSave=now;task.spawn(autoSaveDraft) end
end)

RunService.RenderStepped:Connect(function()
	if mouse and mouse.Hit then updatePreview(getSnappedPos(mouse.Hit))
	else previewPart.Transparency=1;previewBox.Adornee=nil end
end)

	_TB.width=math.clamp(_TB.width-2,2,100);_TB.height=math.clamp(_TB.height-2,2,100);_TB.depth=math.clamp(_TB.depth-2,2,100)
	_TB.currentBlockSize=Vector3.new(_TB.width,_TB.height,_TB.depth)
	UI.widthLabel.Text=tostring(_TB.width);UI.heightLabel.Text=tostring(_TB.height);UI.depthLabel.Text=tostring(_TB.depth);updateHUD()
end)

-- KEYBOARD SHORTCUTS
UIS.InputBegan:Connect(function(input,gpe)
	if gpe then return end
	local k=input.KeyCode
	if k==Enum.KeyCode.E then
		if _TB.flattenMode or _TB.smoothMode or _TB.erodeMode or _TB.growMode or _TB.paintMode or _TB.seaLevelMode then
			if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end
			if _TB.updateSpecialBtns then _TB.updateSpecialBtns() end
			if _TB.updateQuickDisableBtn then _TB.updateQuickDisableBtn() end
		else
			_TB.placementEnabled=not _TB.placementEnabled
			if _TB.placementEnabled and _TB.tunnelMode then _TB.tunnelMode=false;UI.tunnelBtn.Text="Mode Gali: OFF";UI.tunnelBtn.BackgroundColor3=Color3.fromRGB(68,68,100) end
			if _TB.updateQuickDisableBtn then _TB.updateQuickDisableBtn() end
		end
	elseif k==Enum.KeyCode.R then
		local rAngles=UI.rotAngles or {0,45,90,135,180}
		local rIdx=(UI.rotIndex or 1)%#rAngles+1;UI.rotIndex=rIdx;_TB.rotationAngle=rAngles[rIdx];if UI.rotDisplay then UI.rotDisplay.Text=tostring(_TB.rotationAngle).."°" end;updateHUD()
	elseif k==Enum.KeyCode.LeftBracket then
		_TB.width=math.clamp(_TB.width-2,2,100);_TB.height=math.clamp(_TB.height-2,2,100);_TB.depth=math.clamp(_TB.depth-2,2,100)
		_TB.currentBlockSize=Vector3.new(_TB.width,_TB.height,_TB.depth);UI.widthLabel.Text=tostring(_TB.width);UI.heightLabel.Text=tostring(_TB.height);UI.depthLabel.Text=tostring(_TB.depth);updateHUD()
	elseif k==Enum.KeyCode.RightBracket then
		_TB.width=math.clamp(_TB.width+2,2,100);_TB.height=math.clamp(_TB.height+2,2,100);_TB.depth=math.clamp(_TB.depth+2,2,100)
		_TB.currentBlockSize=Vector3.new(_TB.width,_TB.height,_TB.depth);UI.widthLabel.Text=tostring(_TB.width);UI.heightLabel.Text=tostring(_TB.height);UI.depthLabel.Text=tostring(_TB.depth);updateHUD()
	elseif k==Enum.KeyCode.Z then undoBrushTerrain()
	elseif k==Enum.KeyCode.Y then redoBrushTerrain()
	elseif k==Enum.KeyCode.P then
		_TB.panelVisible=not _TB.panelVisible;pluginFrame.Visible=_TB.panelVisible
		UI.toggleLogo.BackgroundColor3=_TB.panelVisible and Color3.fromRGB(35,135,55) or Color3.fromRGB(20,20,28)
	elseif k==Enum.KeyCode.F then
		if _TB.flattenMode then if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end
		else if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end;_TB.flattenMode=true;_TB.placementEnabled=false end
		if _TB.updateSpecialBtns then _TB.updateSpecialBtns() end;if _TB.updateQuickDisableBtn then _TB.updateQuickDisableBtn() end
	elseif k==Enum.KeyCode.G then
		if _TB.growMode then if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end
		else if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end;_TB.growMode=true;_TB.placementEnabled=false end
		if _TB.updateSpecialBtns then _TB.updateSpecialBtns() end;if _TB.updateQuickDisableBtn then _TB.updateQuickDisableBtn() end
	elseif k==Enum.KeyCode.T then
		if _TB.smoothMode then if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end
		else if _TB.disableAllSpecialModes then _TB.disableAllSpecialModes() end;_TB.smoothMode=true;_TB.placementEnabled=false end
		if _TB.updateSpecialBtns then _TB.updateSpecialBtns() end;if _TB.updateQuickDisableBtn then _TB.updateQuickDisableBtn() end
	elseif k==Enum.KeyCode.Tab then
		local tNames=_TB.tabNames or {"ADD","REMOVE","PAINT","TOOL"}
		local idx=1;for i,n in ipairs(tNames) do if n==_TB.activeTab then idx=i;break end end
		if _TB.switchTab then _TB.switchTab(tNames[idx%#tNames+1]) end
	end
end)

-- ═══════════════════════════════════════════════════════════
end
